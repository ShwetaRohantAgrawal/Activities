SELECT CUSTOMER_ID,MAX(PURCHASE_AMOUNT) AS "MAXAMOUNT" FROM ORDERS GROUP BY CUSTOMER_ID;

SELECT CUSTOMER_ID,ORDER_DATE,MAX(PURCHASE_AMOUNT) AS "MAXAMOUNT" FROM ORDERS WHERE ORDER_DATE = TO_DATE('2012/08/17', 'YYYY/MM/DD') GROUP BY CUSTOMER_ID,ORDER_DATE  
;
SELECT CUSTOMER_ID,ORDER_DATE,MAX(PURCHASE_AMOUNT) AS "MAXAMOUNT" FROM ORDERS   GROUP BY CUSTOMER_ID,ORDER_DATE HAVING (MAX(PURCHASE_AMOUNT) IN (2030, 3450, 5760, 6000));
