SELECT * FROM(
SELECT CUSTOMER_ID, CUSTOMER_NAME FROM CUSTOMERS WHERE CUSTOMER_ID  IN (SELECT CUSTOMER_ID FROM ORDERS GROUP BY CUSTOMER_ID HAVING COUNT(1)>1)
UNION
SELECT SALESMAN_ID, SALESMAN_NAME FROM SALESMAN WHERE SALESMAN_ID IN (SELECT SALESMAN_ID FROM ORDERS GROUP BY SALESMAN_ID HAVING COUNT(1)>1)
) ORDER BY CUSTOMER_NAME

;

SELECT S.SALESMAN_ID, ORDER_NO, ORDER_DATE,O.PURCHASE_AMOUNT,'HIGHEST' FROM SALESMAN S, ORDERS O WHERE O.SALESMAN_ID = S.SALESMAN_ID AND O.PURCHASE_AMOUNT = (
SELECT  MAX(PURCHASE_AMOUNT) FROM ORDERS WHERE ORDER_DATE = O.ORDER_DATE)
UNION
SELECT S.SALESMAN_ID, ORDER_NO, ORDER_DATE,O.PURCHASE_AMOUNT,'LOWEST' FROM SALESMAN S, ORDERS O WHERE O.SALESMAN_ID = S.SALESMAN_ID AND O.PURCHASE_AMOUNT = (
SELECT  MIN(PURCHASE_AMOUNT) FROM ORDERS WHERE ORDER_DATE = O.ORDER_DATE)
